lualib
里面是openresty的几个插件 包括
cjson
LuaXML
struct
websocket
# 安装
wget https://github.com/xlaoshu/tools/raw/refs/heads/main/lualib.tar
tar -xf lualib.tar -C /usr/local/openresty
systemctl restart nginx.service
